50// /**
51// * @brief dw1000 spi 头部数据
52// * @note 对于dw1000通信来说，相比于一般的spi设备较为复杂。
53// * 对于一次spi通信，需要先发送一帧头部数据（包含寄存器地址），
54// * 然后才开始发送（接收）相应的数据
55// * @note 头部数据最长为3个字节。
56// * 第一个字节为命令。
57// * bit[7]:读/写，0-读，1-写；bit[6]:子地址，0-无子地址，1-有子地址；bit[5:0]:寄存器地址
58// * 第二个字节为子地址的低7位（如果存在，即命令字节的bit[6]为1）。
59// * bit[7]: 扩展子地址，0-子地址长度为1字节，1-子地址长度为2字节；bit[6:0] 子地址值（如果长度为2字节，该处为子地址的低7位）
60// * 第三个字节为子地址的高7位（如果存在，即第二个字节的bit[7]为1）。
61// * bit[7:0]：为子地址的bit[14:7]的数据

87static uint8_t a_dw1000_spi_read(dw1000_handle_t* handle, uint8_t reg, uint16_t offset, uint8_t* buf, uint16_t len) {
88 #if (DRIVER_DW1000_PARAM_CHECK == 1)
89 assert(reg <= 0x3F);
90 assert(handle != NULL);
91 assert(handle->spi_read != NULL);
92 assert(buf != NULL);
93 assert(len != 0);
94 #endif
95 /* 对输入参数进行检查（这里定义为静态不给外界调用，所以默认调用是正确的，也就省略输入参数检查） */
96 // if (handle == NULL || handle->spi_read == NULL || buf == NULL || len == 0) {
97 // return 1;
98 // }
99 /* 定义头部数组 */
100 uint8_t dw1000_spi_header[3];
101 /* 定义头部数组长度 */
102 uint8_t dw1000_spi_hdr_len;
103 /* 如果偏移为0，即只有寄存器地址，头部字段长度为1字节 */
104 if(offset == 0){
105 /* 设置本次spi为读操作且无子地址 */
106 // dw1000_spi_header[0] = DW1000_SPI_NOT_SUB_ADDR(DW1000_SPI_CMD_READ(reg));
107 dw1000_spi_header[0] = reg;
108 dw1000_spi_hdr_len = 1;
109 }
110 /* 如果偏移不超过127，也就是用7bit可以表示时，说明不需要扩展子地址 */
111 else if(offset <= 0x7F){
112 /* 设置本次spi为读操作且有子地址，但无扩展子地址 */
113 // dw1000_spi_header[0] = DW1000_SPI_HAS_SUB_ADDR(DW1000_SPI_CMD_READ(reg));
114 dw1000_spi_header[0] = DW1000_SPI_HAS_SUB_ADDR(reg);
115 // dw1000_spi_header[1] = DW1000_SPI_NOT_EXT_SUB_ADDR(offset);
116 dw1000_spi_header[1] = offset;
117 dw1000_spi_hdr_len = 2;
118 }
119 else{
120 /* 设置本次spi为读操作且有子地址，且有扩展子地址 */
121 // dw1000_spi_header[0] = DW1000_SPI_HAS_SUB_ADDR(DW1000_SPI_CMD_READ(reg));
122 dw1000_spi_header[0] = DW1000_SPI_HAS_SUB_ADDR(reg);
123 dw1000_spi_header[1] = DW1000_SPI_HAS_EXT_SUB_ADDR(offset);
124 dw1000_spi_header[2] = (offset >> 7);
125 dw1000_spi_hdr_len = 3;
126 }
127 return handle->spi_read(dw1000_spi_header, dw1000_spi_hdr_len, buf, len);
128}
